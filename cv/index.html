Codewars task Is a number prime?

function gap(g, m, n) {
  function isPrime(n){
    if (n%1!=0){return false}
    for (let i=2;i<Math.sqrt(n)+1;){
      if (n%i==0) return false;
      i==2 ? i+=1 : i+=2
    };
    return true;
  };
  function nextPrime(n){
    let i=n+1;
    while(!isPrime(i)){i++}
    return i
  }
  let pr = [nextPrime(m-1),nextPrime(nextPrime(m-1))]
  for(;pr[1]<=n;pr[0]=pr[1]){
    pr[1]=nextPrime(pr[0])
    if (pr[1]-pr[0]==g){return [pr[0],pr[1]]}
  }
  return null
};
